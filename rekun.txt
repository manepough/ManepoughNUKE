-- [[ 1. SAFE LOADSTRING ]] --
local function safeLoad(url)
    local success, code = pcall(game.HttpGet, game, url)
    if not success then warn("Failed to fetch code: " .. tostring(code)) return end
    local executable, compileError = loadstring(code)
    if executable then
        local runSuccess, runError = pcall(executable)
        if not runSuccess then warn("Code ran but crashed: " .. tostring(runError)) end
    else
        warn("Syntax error in loaded code: " .. tostring(compileError))
    end
end

local currentTargetBtn, lastUpdate = nil, 0
local function updatePicker(input)
    if os.clock() - lastUpdate < 0.015 then return end; lastUpdate = os.clock()
    local center = Wheel.AbsolutePosition + (Wheel.AbsoluteSize/2)
    local delta = Vector2.new(input.Position.X, input.Position.Y) - center
    local dist = delta.Magnitude; local radius = Wheel.AbsoluteSize.X / 2
    local sat = math.clamp(dist/radius, 0, 1)
    local hue = (math.atan2(delta.Y, delta.X) + math.pi/2) / (math.pi * 2)
    if hue < 0 then hue = hue + 1 end
    
    local color = Color3.fromHSV(hue, sat, 1)
    if currentTargetBtn then currentTargetBtn.BackgroundColor3 = color end
    RGBInput.Text = string.format("%d, %d, %d", math.round(color.R*255), math.round(color.G*255), math.round(color.B*255))
    SelectorDot.Position = UDim2.new(0.5, (delta.Unit * (sat * radius)).X, 0.5, (delta.Unit * (sat * radius)).Y)
end

RGBInput.FocusLost:Connect(function()
    local r, g, b = string.match(RGBInput.Text, "(%d+)%D+(%d+)%D+(%d+)")
    if r and g and b then
        r, g, b = math.clamp(tonumber(r), 0, 255), math.clamp(tonumber(g), 0, 255), math.clamp(tonumber(b), 0, 255)
        local customColor = Color3.fromRGB(r, g, b)
        if currentTargetBtn then currentTargetBtn.BackgroundColor3 = customColor end
        RGBInput.Text = string.format("%d, %d, %d", r, g, b)
        local h, s, v = Color3.toHSV(customColor)
        local radius = Wheel.AbsoluteSize.X / 2
        local angle = (h * math.pi * 2) - math.pi / 2
        SelectorDot.Position = UDim2.new(0.5, math.cos(angle) * (s * radius), 0.5, math.sin(angle) * (s * radius))
    end
end)

local function runFix(isCleaning)
    local char = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
    local tool = LocalPlayer.Backpack:FindFirstChild("Paint") or char:FindFirstChild("Paint")
    if not tool then return end
    if tool.Parent ~= char then char:WaitForChild("Humanoid"):EquipTool(tool); task.wait(0.4) end
    local remote = tool:FindFirstChild("Event", true) or tool:FindFirstChildWhichIsA("RemoteEvent", true)
    local rootPart = char:FindFirstChild("HumanoidRootPart")
    if not remote or not brick or not rootPart then return end
    
    local rootPos = rootPart.Position
    local key = "both \u{1F91D}"
    
    local mat = isCleaning and "plastic" or (useToxic and "toxic" or "plastic")
    local col = isCleaning and Color3.fromRGB(163, 162, 165) or Color3.new(0,0,0)
    
    -- Step 1: Base & Material (Restored Exact Arguments)
    pcall(function() remote:FireServer(brick, Enum.NormalId.Top, rootPos, key, col, mat, "anchor") end)
    task.wait(0.4)
    
    -- Step 2: Auto Anchor (Restored Exact Arguments)
    if autoAnchor then 
        pcall(function() remote:FireServer(brick, Enum.NormalId.Top, rootPos, key, col, "anchor", "") end)
        task.wait(0.4) 
    end
    
    -- Step 3: Numeric Sides Loop
    for sideNum = 0, 5 do
        local faceName
        for n, id in pairs(nameToId) do if id == sideNum then faceName = n break end end
        local cfg = faceData[faceName]
        local sideEnum = Enum.NormalId:GetEnumItems()[sideNum + 1]
        pcall(function() remote:FireServer(brick, sideEnum, rootPos, key, cfg.clrBtn.BackgroundColor3, "spray", cfg.txtBox.Text) end)
        task.wait(0.2)
    end
end
